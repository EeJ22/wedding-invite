    <script>
        // --- Photo Upload Logic ---
        function previewImage(event, elementId) {
            const reader = new FileReader();
            reader.onload = function(){
                const output = document.getElementById(elementId);
                output.src = reader.result;
            }
            if(event.target.files[0]) {
                reader.readAsDataURL(event.target.files[0]);
            }
        }

        // --- Countdown Logic ---
        const weddingDate = new Date("February 14, 2026 17:30:00").getTime();

        const timer = setInterval(function() {
            const now = new Date().getTime();
            const distance = weddingDate - now;

            const days = Math.floor(distance / (1000 * 60 * 60 * 24));
            const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((distance % (1000 * 60)) / 1000);

            const timerHTML = `
                <div class="bg-amber-800/50 p-4 rounded backdrop-blur-md">
                    <div class="text-3xl font-serif">${days}</div>
                    <div class="text-[10px] uppercase tracking-wider opacity-70">Days</div>
                </div>
                <div class="bg-amber-800/50 p-4 rounded backdrop-blur-md">
                    <div class="text-3xl font-serif">${hours}</div>
                    <div class="text-[10px] uppercase tracking-wider opacity-70">Hours</div>
                </div>
                <div class="bg-amber-800/50 p-4 rounded backdrop-blur-md">
                    <div class="text-3xl font-serif">${minutes}</div>
                    <div class="text-[10px] uppercase tracking-wider opacity-70">Mins</div>
                </div>
                <div class="bg-amber-800/50 p-4 rounded backdrop-blur-md">
                    <div class="text-3xl font-serif">${seconds}</div>
                    <div class="text-[10px] uppercase tracking-wider opacity-70">Secs</div>
                </div>
            `;

            document.getElementById("countdown").innerHTML = timerHTML;

            if (distance < 0) {
                clearInterval(timer);
                document.getElementById("countdown").innerHTML = '<div class="col-span-4 text-2xl font-serif">Happily Married!</div>';
            }
        }, 1000);

        // --- RSVP Modal Logic ---
        function handleRSVP(e) {
            e.preventDefault();
            // In a real app, you would send data to a server here.
            document.getElementById('modal').style.display = 'flex';
            e.target.reset();
        }

        function closeModal() {
            document.getElementById('modal').style.display = 'none';
        }
    </script>
